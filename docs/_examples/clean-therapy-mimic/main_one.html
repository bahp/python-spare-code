43  # -----------------------------
44  # .. note: The closed parameter indicates whether to include
45  #          the first and/or last samples. None will keep both,
46  #          left will keep only start date and right will keep
47  #          also the right date.
48  # Create column with date range
49  data['startdate'] = data.apply(lambda x:
50      pd.date_range(start=x['starttime'],
51                    end=x['stoptime'],
52                    closed='left',         # ignoring right
53                    freq='D') ,axis=1)
54 
55  # Explode such column
56  data = data.explode('startdate')
57 
58  # Create daily therapies
59  aux = data.groupby('startdate') \
60      .apply(lambda x: sorted(x.antibiotic \
61          .str.lower().str.strip().unique().tolist()))
62 
63  # Include missing days
64  aux = aux.asfreq('1D')

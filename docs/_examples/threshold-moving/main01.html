 46  def npv_ppv_from_sens_spec(sens, spec, prev):
 47      """Compute npv and ppv.
 48 
 49      Parameters
 50      ----------
 51      sens: array-like
 52      spec: array-like
 53      prev: float
 54      """
 55      npv = (spec * (1 - prev)) / ((spec * (1 - prev)) + ((1 - sens) * prev))
 56      ppv = (sens * prev) / ((sens * prev) + ((1 - spec) * (1 - prev)))
 57      return npv, ppv

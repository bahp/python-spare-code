 21 def extract_basic_travel_data(pdf_path, start_page, end_page):
 22     """Extract the data from a PDF file.
 23
 24     Ensure that the format is appropriate, ent the column headers match those
 25     included below. Otherwise modify as appropriate.
 26
 27     Parameters
 28     ----------
 29     pdf_path: str
 30         The path to the file.
 31     start_page: int
 32         The start page where the table apperas.
 33     end_page: int
 34         The end page where the table appears.
 35
 36     Returns
 37     -------
 38     """
 39     # Define the headers for essential data
 40     headers = [
 41         "Departure Date/Time", "Arrival Date/Time", "Voyage Code", "In/Out",
 42         "Dep Port", "Arrival Port"
 43     ]
 44
 45     travel_data = []
 46
 47     # Regex pattern to capture essential information
 48     row_pattern = re.compile(
 49         r"(\d{2}/\d{2}/\d{4} \d{2}:\d{2})\s+"  # Departure Date/Time
 50         r"(\d{2}/\d{2}/\d{4} \d{2}:\d{2})\s+"  # Arrival Date/Time
 51         r"(\S+)\s+"                            # Voyage Code
 52         r"(Outbound|Inbound)\s+"               # In/Out
 53         r"(\S+)\s+"                            # Dep Port
 54         r"(\S+)"                               # Arrival Port
 55     )
 56
 57     # Open the PDF file and iterate over specified pages
 58     with pdfplumber.open(pdf_path) as pdf:
 59         for page_num in range(start_page - 1, end_page):
 60             page = pdf.pages[page_num]
 61             text = page.extract_text()
 62             if not text:
 63                 continue
 64
 65             # Match rows using the regex pattern
 66             matches = row_pattern.findall(text)
 67             if matches:
 68                 for match in matches:
 69                     travel_data.append(dict(zip(headers, match)))
 70
 71     # Return
 72     return travel_data

 75 def combine_outbound_inbound(df):
 76     """Combine Outbound-Ibound rows into a single one.
 77
 78     Paramters
 79     ---------
 80     df: pd.DataFrame
 81         The DataFrame with the data.
 82
 83     Returns
 84     -------
 85     pd.DataFrame
 86
 87     """
 88
 89     # Convert date columns to datetime format
 90     df["Departure Date/Time"] = \
 91         pd.to_datetime(df["Departure Date/Time"],
 92             format="%d/%m/%Y %H:%M")
 93     df["Arrival Date/Time"] = \
 94         pd.to_datetime(df["Arrival Date/Time"],
 95             format="%d/%m/%Y %H:%M")
 96
 97     # Sort the DataFrame by "Departure Date/Time"
 98     df = df.sort_values(by="Departure Date/Time").reset_index(drop=True)
 99
100     # Process the DataFrame
101     result = []
102     for i in range(0, len(df) - 1, 2):  # Step by 2 to handle consecutive rows
103         outbound = df.iloc[i]
104         inbound = df.iloc[i + 1]
105
106         # Ensure the pair consists of an outbound followed by an inbound
107         if outbound["In/Out"] == "Outbound" and inbound["In/Out"] == "Inbound":
108             # Calculate the difference in days
109             days_difference = (inbound["Arrival Date/Time"] - outbound["Departure Date/Time"]).days - 1
110
111             # Create a combined row with desired columns
112             combined_row = {
113                 "Outbound Date": outbound["Departure Date/Time"],
114                 "Inbound Date": inbound["Arrival Date/Time"],
115                 "Outbound Ports": outbound["Dep Port"] + '-' + outbound["Arrival Port"],
116                 "Inbound Ports": inbound["Dep Port"] + '-' + inbound["Arrival Port"],
117                 "Days Difference": days_difference,
118                 "Voyage Code": outbound["Voyage Code"]
119             }
120
121             result.append(combined_row)
122
123     # Return
124     return pd.DataFrame(result)

128 def display(df, cmap=None):
129     """Plotting the graph.
130
131     Parameters
132     ----------
133     df: pd.DataFrame
134         The pandas DataFrame.
135
136     Returns
137     -------
138     None
139     """
140     # Set up plot
141     fig, ax = plt.subplots(figsize=(16, 8))
142
143     # Fore each row (voyage)
144     for i, row in df.iterrows():
145
146         if cmap is None:
147             color = 'skyblue'
148         else:
149             cmap.get(row['Outbound Ports'].split('-')[1], 'skyblue')
150
151
152         # Plot each voyage as a horizontal bar with text annotations
153         ax.plot([row["Outbound Date"], row["Inbound Date"]], [i, i], marker='o', color=color, lw=6)
154
155         # Formatting outbound and inbound dates
156         outbound_str = row["Outbound Date"].strftime("%d %b")  # Day and abbreviated month
157         inbound_str = row["Inbound Date"].strftime("%d %b")    # Day and abbreviated month
158
159         # Adjust the text position to be further right
160         ax.text(row["Inbound Date"] + pd.Timedelta(days=10), i - 0.05,  # Increased offset to 10 days
161                 f"{row['Outbound Ports']} ({outbound_str}) to {row['Inbound Ports']} ({inbound_str}) | {row['Days Difference']} days",
162                 va='center', ha='left', fontsize=9, color="black")
163
164     # Alternate month shading
165     start_date = df["Outbound Date"].min().replace(day=1)
166     end_date = df["Inbound Date"].max()
167     current_date = start_date
168     month = 0
169     while current_date < end_date:
170         next_month = (current_date + pd.DateOffset(months=1)).replace(day=1)
171         ax.axvspan(current_date, next_month, color='gray' if month % 2 == 0 else 'lightgray', alpha=0.2)
172         current_date = next_month
173         month += 1
174
175     # Add horizontal lines for each year
176     years = pd.date_range(start=start_date, end=end_date+pd.DateOffset(years=1), freq='Y')
177     for year in years:
178         ax.axvline(year, color='black', linestyle='--', lw=1)  # Vertical line for each year
179         ax.text(year - pd.Timedelta(days=90), len(df) + 0.5, year.year,
180             ha='left', va='center', fontsize=10, color='black')  # Year label
181
182     # Setting the x-axis limits to include full years
183     full_start_date = pd.Timestamp(year=start_date.year, month=1, day=1)
184     full_end_date = pd.Timestamp(year=end_date.year + 1, month=1, day=1)  # Next January
185     ax.set_xlim(full_start_date, full_end_date)
186
187     # Set x-axis ticks to show full years from January to December
188     ax.xaxis.set_major_locator(mdates.YearLocator())   # Major ticks at the beginning of each year
189     ax.xaxis.set_minor_locator(mdates.MonthLocator())  # Minor ticks for each month
190     ax.xaxis.set_major_formatter(DateFormatter("%Y"))  # Year as the format for major ticks
191
192     # Formatting the plot
193     ax.set_yticks(range(len(df)))
194     ax.set_yticklabels(df["Voyage Code"])
195     #ax.set_yticklabels(df['Days Difference'])
196     ax.set_xlabel("Date")
197     ax.set_title("Voyage Durations (total abroad %s days)" % df['Days Difference'].sum())
198
199     # Set x-axis ticks to show abbreviated month names and year
200     ax.xaxis.set_major_locator(mdates.MonthLocator())
201     ax.xaxis.set_major_formatter(DateFormatter("%b %Y"))  # Month abbreviation and year
202     plt.xticks(rotation=45)
203
204     plt.grid(axis='x', linestyle='--', alpha=0.5)
205     plt.tight_layout()
206     plt.show()

 20 class DataframeXYWrapper():
 21
 22     def __init__(self, dataframe, X_columns, y_columns):
 23         """
 24         """
 25         self.dataframe = dataframe
 26         self.X_cols = X_columns
 27         self.y_cols = y_columns
 28
 29     def get_X(self):
 30         """Return..."""
 31         return self.dataframe[self.X_cols]
 32
 33     def get_y(self):
 34         """Return..."""
 35         return self.dataframe[self.y_cols]
 36
 37     def get_X_y(self):
 38         """Return...
 39         """
 40         return self.dataframe[self.X_cols + self.y_cols]
 41
 42     def apply(self, obj, func, requires_X=False,
 43                                requires_y=False,
 44                                argname_X='X',
 45                                argname_y='y',
 46                                inplace=True,
 47                                return_obj=True,
 48                                **kwargs):
 49         """
 50         """
 51         if requires_X:
 52             kwargs[argname_X] = self.get_X()
 53         if requires_y:
 54             kwargs[argname_y] = self.get_y()
 55
 56         # Output
 57         out = getattr(obj, func)(**kwargs)
 58
 59         # Inplace
 60         if inplace:
 61             self.dataframe[self.X_cols] = out
 62
 63         # Return
 64         if return_obj:
 65             return obj, out
 66         return out

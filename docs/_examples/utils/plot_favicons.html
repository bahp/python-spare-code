Plot faviconsÔÉÅ

This example lists all the libraries installed in the environment using pip, gets their site url, downloads the icon and displays it in the example.

Note

The command cut is not available in windows! It might available on Cygwin o Git for windows.

 20 class IconHTML:
 21     """Class to display html in sphinx-gallery."""
 22     TMP0 = '<img src={url} style="{s}", width={w} height={h}>'
 23     TMP1 = '<div>'+TMP0+' <span>{name}</span></div>'
 24
 25     """Class to display icons on sphinx-gallery."""
 26     def __init__(self, d, width=25, height=25, verbose=0):
 27         self.d = d
 28         self.width = width
 29         self.height = height
 30         self.style = "display: inline; vertical-align:middle;"
 31         self.verbose = verbose
 32
 33     def _repr_html_short_(self):
 34         return ' '.join([self.TMP0.format(url=v,
 35             w=self.width, h=self.height, s=self.style)
 36                 for k,v in self.d.items()])
 37
 38     def _repr_html_long_(self):
 39         return ' '.join([self.TMP1.format(url=v,
 40             w=self.width, h=self.height, s=self.style, name=k.lower())
 41                 for k, v in self.d.items()])
 42
 43     def _repr_html_(self):
 44         if self.verbose == 0:
 45             return self._repr_html_short_()
 46         return self._repr_html_long_()

 45 # Create timedelta
 46 data['timedelta'] = \

 49 # Create datetimes (if needed)
 50 today = pd.to_datetime('today').normalize()

 52 data['dates'] += pd.to_timedelta(data.index / 1, unit='D')
 53
 54 # Set dates as index
 55 data = data.set_index('dates')

 89 # Add traces

104     # Update axes

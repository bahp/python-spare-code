 36 # Create data

 40 # Create DataFrame
 41 data = pd.DataFrame(y)
 42 data.columns = ['c%s'%i for i in data.columns]
 43
 44 # Create timedelta
 45 data['timedelta'] = \
 46     pd.to_timedelta(data.index / 1, unit='D')
 47
 48 # Create datetimes (if needed)
 49 today = pd.to_datetime('today').normalize()
 50 data['dates'] = pd.to_datetime(today)
 51 data['dates'] += pd.to_timedelta(data.index / 1, unit='D')
 52
 53 # Set dates as index
 54 data = data.set_index('dates')
 55
 56 # Drop timedelta
 57 data = data.drop(columns='timedelta')

 87 # Add traces

102     # Update axes

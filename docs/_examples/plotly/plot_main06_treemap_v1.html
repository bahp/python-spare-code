 # Display
 fig = go.Figure(go.Treemap(
     ids=df.ids,
     labels=df.labels,
     parents=df.parents,
     pathbar_textfont_size=15,
     #maxdepth=3,
     root_color="lightgrey"
 ))

 # Update
 fig.update_layout(
     uniformtext=dict(minsize=10, mode='hide'),
     margin=dict(t=50, l=25, r=25, b=25)
 )

 # Show
 if TERMINAL:
     fig.show()

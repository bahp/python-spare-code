Let’s display a heatmap

 81 # Draw (heatmap)

 92 r2 = sns.heatmap(data=data, cmap=cmapl, mask=maskl, ax=axs,
 93             annot=False, linewidth=0.5, vmin=-0.7, vmax=0.7,
 94             center=0, annot_kws={"size": 8}, square=True,
 95             xticklabels=True, yticklabels=True,
 96             cbar_kws={'label': 'Collateral Resistance Index'})
 97
 98 # Create patches for categories
 99 category_patches = []
100 for i in axs.get_xticklabels():
101     try:
102         x, y = i.get_position()
103         c = colors.to_dict().get(i.get_text(), 'k')
104         #i.set_color(c) # for testing
105
106         # Add patch.
107         category_patches.append(
108             Rectangle((x-0.35, y-0.5), 0.8, 0.3, edgecolor='k',
109                 facecolor=c, fill=True, lw=0.25, alpha=0.5, zorder=1000,
110                 transform=axs.transData
111             )
112         )
113     except Exception as e:
114         print(i.get_text(), e)
115
116 # Add category rectangles
117 fig.patches.extend(category_patches)

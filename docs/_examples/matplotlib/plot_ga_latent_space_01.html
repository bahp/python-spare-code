 34  def make_colormap(seq):
 35      """Return a LinearSegmentedColormap
 36 
 37      Parameters
 38      ----------
 39      seq: list
 40          A sequence of floats and RGB-tuples. The floats
 41          should be increasing and in the interval (0,1).
 42      """
 43      # Library
 44      import matplotlib.colors as mcolors
 45      # Code
 46      seq = [(None,) * 3, 0.0] + list(seq) + [1.0, (None,) * 3]
 47      cdict = {'red': [], 'green': [], 'blue': []}
 48      for i, item in enumerate(seq):
 49          if isinstance(item, float):
 50              r1, g1, b1 = seq[i - 1]
 51              r2, g2, b2 = seq[i + 1]
 52              cdict['red'].append([item, r1, r2])
 53              cdict['green'].append([item, g1, g2])
 54              cdict['blue'].append([item, b1, b2])
 55      return mcolors.LinearSegmentedColormap('CustomMap', cdict)

 79  def kde_mpl_compute(x, y, xlim=None, ylim=None, **kwargs):
 80      """Computes the gaussian kde.
 81 
 82      Parameters
 83      ----------
 84 
 85      Returns
 86      -------
 87      """
 88      try:
 89          # Plot density
 90          kde = gaussian_kde(np.vstack((x, y)), **kwargs)
 91      except Exception as e:
 92          print("Exception! %s" % e)
 93          return None, None, None
 94 
 95      # Parameters
 96      xmin, xmax = min(x), max(x)
 97      ymin, ymax = min(y), max(y)
 98 
 99      # Set xlim and ylim
100      if xlim is not None:
101          xmin, xmax = xlim
102      if ylim is not None:
103          ymin, ymax = ylim
104 
105      # evaluate on a regular grid
106      xgrid = np.linspace(xmin, xmax, 100)
107      ygrid = np.linspace(ymin, ymax, 100)
108      Xgrid, Ygrid = np.meshgrid(xgrid, ygrid)
109      zgrid = kde.evaluate(np.vstack([
110          Xgrid.ravel(),
111          Ygrid.ravel()
112      ]))
113      Zgrid = zgrid.reshape(Xgrid.shape)
114 
115      # Return
116      return xgrid, ygrid, Zgrid

118  def plot_ellipses(gmm, ax, color, n=None):
119      """Plot ellipses from GaussianMixtureModel"""
120 
121      # Define color
122      if color is None:
123          color = 'blue'
124      if n is None:
125          n = 1
126 
127      # Get covariances
128      if gmm.covariance_type == 'full':
129          covariances = gmm.covariances_[n][:2, :2]
130      elif gmm.covariance_type == 'tied':
131          covariances = gmm.covariances_[:2, :2]
132      elif gmm.covariance_type == 'diag':
133          covariances = np.diag(gmm.covariances_[n][:2])
134      elif gmm.covariance_type == 'spherical':
135          covariances = np.eye(gmm.means_.shape[1]) * gmm.covariances_[n]
136 
137      # Compute
138      v, w = np.linalg.eigh(covariances)
139      # v = 2. * np.sqrt(2.) * np.sqrt(v) # Oliver
140      u = w[0] / np.linalg.norm(w[0])
141      angle = np.arctan2(u[1], u[0])
142      angle = 180 * angle / np.pi  # convert to degrees
143      v = 2. * np.sqrt(2.) * np.sqrt(v)
144 
145      # Plot
146      ell = mpl.patches.Ellipse(gmm.means_[n, :2],
147          v[0], v[1], 180 + angle, color=color)
148      ell.set_clip_box(ax.bbox)
149      ell.set_alpha(0.25)
150      ax.add_artist(ell)
151      ax.set_aspect('equal', 'datalim')

 21  def _tp_fp_tn_fn_distributions(y, y_pred, y_prob):
 22      """This function returns probabilities for each of the confusion
 23      matrix elements (tp, tn, fp, fn).
 24 
 25      Parameters
 26      ----------
 27      y : array-like
 28        The real categories
 29 
 30      y_pred : array-like

 35

 38      tp_probs, tn_probs, fp_probs, fn_probs

 40      # Tags.
 41      tp_idx = (y_pred == 1) & (y == 1)
 42      tn_idx = (y_pred == 0) & (y == 0)
 43      fp_idx = (y_pred == 1) & (y == 0)
 44      fn_idx = (y_pred == 0) & (y == 1)

 47      tn_probs = y_prob[tn_idx]
 48      fp_probs = y_prob[fp_idx]

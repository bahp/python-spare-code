This example shows how to plot the probability distributions for each of the components of the confusion matrix. Note that data is created artificially. If necessary, it is possible to limit the axis to the range [0, 1] to keep only those true probability values.



 12 # Libraries
 13 import plotly.graph_objects as go
 14 from plotly.graph_objects import Layout
 15 import plotly.express as px
 16 import pandas as pd
 17 import numpy as np
 18
 19 TERMINAL = False

 21 # -----------------------------------------
 22 # Helper method
 23 # -----------------------------------------
 24 # This method is implemented in pySML.
 25 def _tp_fp_tn_fn_distributions(y, y_pred, y_prob):
 26     """This function returns probabilities for each of the confusion
 27     matrix elements (tp, tn, fp, fn).
 28
 29     Parameters
 30     ----------
 31     y : array-like
 32       The real categories
 33
 34     y_pred : array-like
 35       The predicted categories
 36
 37     y_prob: array-like
 38       The predict probabilities
 39
 40     Returns
 41     -------
 42     tp_probs, tn_probs, fp_probs, fn_probs
 43     """
 44     # Tags.
 45     tp_idx = (y_pred == 1) & (y == 1)
 46     tn_idx = (y_pred == 0) & (y == 0)
 47     fp_idx = (y_pred == 1) & (y == 0)
 48     fn_idx = (y_pred == 0) & (y == 1)
 49     # Show information.
 50     tp_probs = y_prob[tp_idx]
 51     tn_probs = y_prob[tn_idx]
 52     fp_probs = y_prob[fp_idx]
 53     fn_probs = y_prob[fn_idx]
 54     # Return
 55     return tp_probs, tn_probs, fp_probs, fn_probs

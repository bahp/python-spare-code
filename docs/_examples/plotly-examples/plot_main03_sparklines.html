 21  from plotly.subplots import make_subplots

 26  except:
 27      TERMINAL = False
 28 
 29  # Constants
 30  colors = px.colors.sequential.Viridis_r
 31 
 32  # Size
 33  S = 100
 34  N = 7
 35 
 36  # Create data
 37  x = np.arange(S)
 38  y = np.random.randint(low=1, high=100, size=(S, N))
 39 
 40  # Create DataFrame
 41  data = pd.DataFrame(y)
 42  data.columns = ['c%s'%i for i in data.columns]
 43 
 44  # Create timedelta
 45  data['timedelta'] = \
 46      pd.to_timedelta(data.index / 1, unit='D')
 47 
 48  # Create datetimes (if needed)
 49  today = pd.to_datetime('today').normalize()
 50  data['dates'] = pd.to_datetime(today)
 51  data['dates'] += pd.to_timedelta(data.index / 1, unit='D')
 52 
 53  # Set dates as index
 54  data = data.set_index('dates')
 55 
 56  # Drop timedelta
 57  data = data.drop(columns='timedelta')
 58 
 59  # Show data
 60  if TERMINAL:
 61      print("\nData:")
 62      print(data)
 63 
 64 
 65

 67  # Visualize
 68  # ----------------

 70  layout = {
 71    "font": {"family": "Georgia, serif"},

 74    "height": 500,
 75    "margin": {"t": 80},
 76    "paper_bgcolor": 'rgba(0,0,0,0)',  # transparent
 77    "plot_bgcolor": 'rgba(0,0,0,0)',  # transparent
 78    "autosize": False,
 79    "hovermode": "closest",
 80    "showlegend": False,
 81  }
 82 
 83  # Create figure
 84  fig = make_subplots(rows=N, cols=1,

 87  # Add traces
 88  for i, column in enumerate(data.columns):
 89      # Colors
 90      c = colors[i]
 91      x = data.index
 92      y = data[column]
 93 
 94      # Add trace

 96          name=column.upper(),
 97          mode='lines', fill='tozeroy',
 98          line=dict(color=c, width=0.5),
 99          xaxis='x%s' % (i+1), yaxis='y%s' % (i+1)),

101 
102      # Update axes

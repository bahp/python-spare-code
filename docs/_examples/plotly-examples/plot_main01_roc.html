 32  colors = px.colors.qualitative.Plotly
 33  colors = px.colors.sequential.Plasma_r
 34  colors = px.colors.sequential.Viridis_r
 35 
 36  # -----------------------------------------
 37  # Data
 38  # -----------------------------------------
 39  # Create some data
 40  fpr = np.arange(10)/10
 41  tpr = np.arange(10)/10
 42 
 43  # Data
 44  data = {
 45      'split1': np.vstack((fpr + -0.10, tpr)).T,
 46      'split2': np.vstack((fpr + -0.15, tpr)).T,
 47      'split3': np.vstack((fpr + -0.20, tpr)).T,
 48      'split4': np.vstack((fpr + -0.25, tpr)).T,
 49      'split5': np.vstack((fpr + -0.30, tpr)).T,
 50      'split6': np.vstack((fpr + 0.15, tpr)).T,
 51      'split7': np.vstack((fpr + 0.15, tpr)).T,
 52      'split8': np.vstack((fpr + 0.20, tpr)).T,
 53      'split9': np.vstack((fpr + 0.25, tpr)).T,
 54      'split10': np.vstack((fpr + 0.30, tpr)).T
 55  }
 56 
 57 
 58  # -------------------------------------
 59  # Visualize
 60  # -------------------------------------
 61  # Create figure
 62  fig = go.Figure()
 63 
 64  # Add diagonal line
 65  fig.add_shape(type='line', x0=0, x1=1, y0=0, y1=1,
 66      line=dict(dash='dash', color='gray', width=1),
 67  )
 68 
 69  # Plot each split
 70  for i, (name, array) in enumerate(data.items()):
 71      # Name of split
 72      name = f"{name}" # (AUC={10:.2f})"
 73      # Add trace
 74      fig.add_trace(go.Scatter(x=array[:, 0],
 75                               y=array[:, 1],
 76                               name=name,
 77                               mode='lines+markers',
 78                               line=dict(color=colors[i],
 79                                         width=0.5)))
 80 
 81  # Update layout
 82  fig.update_layout(
 83      xaxis_title='False Positive Rate',
 84      yaxis_title='True Positive Rate',
 85      yaxis=dict(scaleanchor="x", scaleratio=1),
 86      xaxis=dict(constrain='domain'),
 87      width=350, height=350,
 88      legend=dict(
 89          x=1.0, y=0.0,  # x=1, y=1.02
 90          orientation="v",
 91          font={'size': 12},
 92          yanchor="bottom",
 93          xanchor="right",

 95      margin={

103      plot_bgcolor='rgba(0,0,0,0)'  # transparent
104  )

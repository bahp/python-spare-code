 18 def build_hierarchical_dataframe(df, levels, value_column, color_columns=None):
 19     """
 20     Build a hierarchy of levels for Sunburst or Treemap charts.
 21
 22     Levels are given starting from the bottom to the top of the hierarchy,
 23     ie the last level corresponds to the root.
 24     """
 25     df_all_trees = pd.DataFrame(columns=['id', 'parent', 'value', 'color'])
 26     for i, level in enumerate(levels):
 27         df_tree = pd.DataFrame(columns=['id', 'parent', 'value', 'color'])
 28         dfg = df.groupby(levels[i:]).sum()
 29         dfg = dfg.reset_index()
 30         df_tree['id'] = dfg[level].copy()
 31         if i < len(levels) - 1:
 32             df_tree['parent'] = dfg[levels[i+1]].copy()
 33         else:
 34             df_tree['parent'] = 'total'
 35         df_tree['value'] = dfg[value_column]
 36         #df_tree['color'] = dfg[color_columns[0]] / dfg[color_columns[1]]
 37         df_all_trees = df_all_trees.append(df_tree, ignore_index=True)
 38     #total = pd.Series(dict(id='total', parent='',
 39     #                          value=df[value_column].sum(),
 40     #                          color=df[color_columns[0]].sum() / df[color_columns[1]].sum()))
 41     total = pd.Series(dict(id='total', parent='', value=df[value_column].sum()))
 42     df_all_trees = df_all_trees.append(total, ignore_index=True)
 43     return df_all_trees

 55 def load_microbiology_nhs(n=10000):
 56     """Loads and formats microbiology data."""
 57     # Libraries
 58     from pyamr.core.sari import sari
 59     from pyamr.datasets.load import load_data_nhs
 60
 61     # Load data
 62     data, antimicrobials, microorganisms = \
 63         load_data_nhs(nrows=n)
 64
 65     data = data[data.specimen_code.isin(['BLOOD CULTURE'])]
 66
 67     # Create DataFrame
 68     dataframe = data.groupby(['specimen_code',
 69                               'microorganism_code',
 70                               'antimicrobial_code',
 71                               'sensitivity']) \
 72         .size().unstack().fillna(0)
 73
 74
 75     # Compute frequency
 76     dataframe['freq'] = dataframe.sum(axis=1)
 77     dataframe['sari'] = sari(dataframe, strategy='hard')
 78     dataframe['sari_medium'] = sari(dataframe, strategy='medium')
 79     dataframe['sari_soft'] = sari(dataframe, strategy='soft')
 80     dataframe = dataframe.reset_index()
 81
 82     # Add info for popup (micro and abxs)
 83     dataframe = dataframe.merge(antimicrobials,
 84         how='left', left_on='antimicrobial_code',
 85         right_on='antimicrobial_code')
 86     dataframe = dataframe.merge(microorganisms,
 87         how='left', left_on='microorganism_code',
 88         right_on='microorganism_code')
 89
 90     # Format dataframe
 91     dataframe = dataframe.round(decimals=3)
 92
 93     # Configuration
 94     LEVELS = ['specimen_code', 'microorganism_code', 'antimicrobial_code']
 95     COLORS = ['sari']
 96     VALUE = 'freq'
 97
 98     dataframe = dataframe[LEVELS + COLORS + [VALUE]]
 99
100     aux2 = dataframe.groupby(LEVELS).agg('sum').reset_index()
101
102
103     # Return
104     aux = build_hierarchical_dataframe(aux2, LEVELS, COLORS, VALUE)
105
106     return aux

             'dn': 'dc=users,dc=zoldar,dc=net',

             'dn': 'o=foocompany,dc=users,dc=zoldar,dc=net',

        ]
    })

    try:
        server.start()

        pw = "pass1"

        srv = ldap3.Server('localhost', port=3333)

        base_dn = 'dc=zoldar,dc=net'
